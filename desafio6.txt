function convertToHex(cor){
	switch (cor){		
		case 'branco':
			return "O hexadecimal para a cor "+cor+" é fff"
		break;
		case 'preto':
			return "O hexadecimal para a cor "+cor+" é 000"
		break;
		case 'vermelho':
			return "O hexadecimal para a cor "+cor+" é #ff0000"
		break;
		case 'verde':
			return "O hexadecimal para a cor "+cor+" é #00fe00"
		break;
		case 'azul':
			return "O hexadecimal para a cor "+cor+" é #0001ff"
		break;
		default:
			return "Não temos o equivalente hexadecimal para "+cor
	}
}
convertToHex('branco') // "O hexadecimal para a cor branco é fff"
convertToHex('preto') // "O hexadecimal para a cor preto é 000"
convertToHex('vermelho') // "O hexadecimal para a cor vermelho é #ff0000"
convertToHex('azul') // "O hexadecimal para a cor azul é #0001ff"
convertToHex('verde') // "O hexadecimal para a cor verde é #00fe00"
convertToHex('amarelo') // "Não temos o equivalente hexadecimal para amarelo"
convertToHex('violeta') // "Não temos o equivalente hexadecimal para violeta"
convertToHex('magenta') // "Não temos o equivalente hexadecimal para magenta"
